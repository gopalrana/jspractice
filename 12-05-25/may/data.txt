function isArmstrong(num) {
  let strNum = num.toString();
  let n = strNum.length;
  let sum = 0;

  for (let char of strNum) {
    sum += Math.pow(parseInt(char), n);
  }

```js
function bubbleSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        // swap
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
  return arr;
}
```

### 52. **Selection Sort**

```js
function selectionSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n - 1; i++) {
    let minIndex = i;
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[minIndex]) {
        minIndex = j;
      }
    }
    [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
  }
  return arr;
}
```

```js
function insertionSort(arr) {
  let n = arr.length;
  for (let i = 1; i < n; i++) {
    let key = arr[i];
    let j = i - 1;
    while (j >= 0 && arr[j] > key) {
      arr[j + 1] = arr[j];
      j--;
    }
    arr[j + 1] = key;
  }
  return arr;
}
```

### 54. **Merge Sort**

```js
function mergeSort(arr) {
  if (arr.length <= 1) return arr;

  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));

  return merge(left, right);
}

function merge(left, right) {
  let result = [], i = 0, j = 0;

  while (i < left.length && j < right.length) {
    if (left[i] < right[j]) {
      result.push(left[i++]);
    } else {
      result.push(right[j++]);
    }
  }
  return result.concat(left.slice(i)).concat(right.slice(j));
}
```

### 55. **Quick Sort**

```js
function quickSort(arr) {
  if (arr.length <= 1) return arr;

  const pivot = arr[arr.length - 1];
  const left = [], right = [];

  for (let i = 0; i < arr.length - 1; i++) {
    if (arr[i] < pivot) left.push(arr[i]);
    else right.push(arr[i]);
  }

  return [...quickSort(left), pivot, ...quickSort(right)];
}
```

```js
function binarySearch(arr, target) {
  let left = 0, right = arr.length - 1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) return mid;
    else if (arr[mid] < target) left = mid + 1;
    else right = mid - 1;
  }

  return -1; // not found
}
```

### 58. **Counting Sort** (Assuming all values are non-negative integers)

```js
function countingSort(arr) {
  const max = Math.max(...arr);
  const count = new Array(max + 1).fill(0);

  for (let num of arr) {
    count[num]++;
  }

  let sortedIndex = 0;
  for (let i = 0; i <= max; i++) {
    while (count[i] > 0) {
      arr[sortedIndex++] = i;
      count[i]--;
    }
  }
  return arr;
}
```

```js
function firstOccurrence(arr, target) {
  let left = 0, right = arr.length - 1;
  let result = -1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (arr[mid] === target) {
      result = mid;
      right = mid - 1;  // keep searching left side
    } else if (arr[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  return result;
}
```

    function areAnagrams(str1, str2) {
      // Convert both strings to lowercase
      str1 = str1.toLowerCase();
      str2 = str2.toLowerCase();
   
      // Early exit if lengths don't match
      if (str1.length !== str2.length) return false;
   
     
      // Create frequency counter for str1
      const count = {};
   
      for (let i = 0; i < str1.length; i++) {
          const char = str1[i];
          count[char] = (count[char] || 0) + 1;
      }
   
      // Decrease frequency based on str2
      for (let i = 0; i < str2.length; i++) {
          const char = str2[i];
          if (!count[char]) {
              return false; // char not found or used up
          }
          count[char]--;
      }
   
      return true; // all counts are zero
  }

  console.log(areAnagrams("listen", "silent"));     // true

## üî∑ **3. How to scale a Node.js backend?**

### ‚úÖ Tips to scale:

1. **Use Clustering:**

   * Use multiple CPU cores (`cluster` module or PM2)

2. **Use Load Balancer:**

   * Nginx or AWS ELB to distribute load across servers

3. **Optimize DB:**

   * Use indexes, avoid unnecessary queries, pagination

4. **Use Caching:**

   * Redis for session/token caching, frequently accessed data

5. **Use Message Queues:**

   * For background tasks (RabbitMQ, Bull with Redis)

6. **Dockerize Your App:**

   * Use containers for consistent deployment

7. **Break into Microservices (if large):**

   * Separate auth, posts, uploads, etc.

### ‚ùì **JWT Authentication ‡§ï‡•à‡§∏‡•á implement ‡§ï‡§∞‡§§‡•á ‡§π‡•à‡§Ç?**

üìå JWT (JSON Web Token) ‡§ï‡§æ use ‡§π‡•ã‡§§‡§æ ‡§π‡•à **authentication** ‡§ï‡•á ‡§≤‡§ø‡§è ‚Äì ‡§Æ‡§§‡§≤‡§¨ user ‡§ï‡•ã verify ‡§ï‡§∞‡§®‡•á ‡§ï‡•á ‡§≤‡§ø‡§è‡•§

#### Steps:

1. **Login Route** ‚Äì User ‡§∏‡•á credentials ‡§≤‡•ã ‡§î‡§∞ JWT ‡§¨‡§®‡§æ‡§ì:

```js

  const user = { id: 1 }; // DB ‡§∏‡•á validate ‡§ï‡§∞‡•ã
  const token = jwt.sign(user, "SECRET_KEY", { expiresIn: "1h" });
  res.json({ token });
});
```

2. **Middleware to protect routes**:

```js
const authenticate = (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.sendStatus(401);

  try {
    const decoded = jwt.verify(token, "SECRET_KEY");
    req.user = decoded;
    next();
  } catch (err) {
    res.sendStatus(403);
  }
};
```

3. **Protected Route**:

```js
app.get("/dashboard", authenticate, (req, res) => {
  res.send("Welcome, user " + req.user.id);
});
```

## üß† What is the **Event Loop**?

JavaScript ‡§è‡§ï **single-threaded** language ‡§π‡•à ‚Äî ‡§á‡§∏‡§ï‡§æ ‡§Æ‡§§‡§≤‡§¨ ‡§π‡•à ‡§è‡§ï time ‡§™‡§∞ ‡§è‡§ï ‡§π‡•Ä task run ‡§π‡•ã ‡§∏‡§ï‡§§‡§æ ‡§π‡•à. ‡§≤‡•á‡§ï‡§ø‡§® ‡§´‡§ø‡§∞ ‡§≠‡•Ä JS asynchronous ‡§ö‡•Ä‡§ú‡§º‡•á‡§Ç ‡§ï‡•à‡§∏‡•á handle ‡§ï‡§∞‡§§‡§æ ‡§π‡•à?

üëâ ‡§á‡§∏‡§ï‡§æ ‡§ú‡§µ‡§æ‡§¨ ‡§π‡•à: **Event Loop**
It allows JS to perform **non-blocking** operations using:

* Call Stack
* Web APIs (like `setTimeout`, `fetch`)
* Callback Queue / Task Queue
* Microtask Queue (Promises)
* Event Loop

## üí° Tips for Interviews

* Promises are **microtasks**, `setTimeout` is a **macrotask**
* Event Loop always clears **microtasks before** moving to next macrotask
* JS is single-threaded, but browser APIs allow async behavior
* React batching and effects use Event Loop mechanics under the hood

* `next(err)`: Triggers the error-handling middleware.

### üßµ 6. **Explain the lifecycle of a request in Express.**

1. Request hits Express server
2. Middlewares run in order (like body-parser, logger, etc.)
3. Matched route handler runs
4. `res.send()` or `res.json()` sends the response
5. If error: Error-handling middleware runs

---

### üì¶ 7. **What is the difference between `require()` and `import` in Node.js?**

* `require()` is CommonJS
* `import` is ESModule (only in latest Node versions with `"type": "module"`)

> Follow-up: "How do you mix both? Can you use `__dirname` with ESModules?"

### ‚è±Ô∏è 8. **How to handle async errors in Express routes?**

Wrong way:

```js
app.get('/', async (req, res) => {
  const user = await getUser(); // If throws, unhandled
  res.send(user);
});
```

‚úÖ Correct:

```js
app.get('/', asyncHandler(async (req, res) => {
  const user = await getUser();
  res.send(user);
}));
```

Use a wrapper like:

```js
const asyncHandler = fn => (req, res, next) =>
  Promise.resolve(fn(req, res, next)).catch(next);
```

### üîÅ 9. **What is the difference between `process.nextTick()` and `setImmediate()`?**

| Function             | Executes                                      |
| -------------------- | --------------------------------------------- |
| `process.nextTick()` | Before event loop continues (higher priority) |
| `setImmediate()`     | After I/O events, in the next tick            |

### üß™ 10. **What is the difference between `res.end()`, `res.send()`, and `res.json()`?**

* `res.end()`: Ends the response without data (or with raw data)
* `res.send()`: Sends text, buffer, or object
* `res.json()`: Sends object with proper JSON headers

### üîê 3. **Authentication Best Practice**

* Use `POST /auth/login` and return **JWT token**
* Protect private routes using middleware

```js
// Middleware example
function verifyToken(req, res, next) {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(401).json({ message: "Unauthorized" });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ message: "Invalid token" });
  }
}
```
